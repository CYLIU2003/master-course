### 3.1 タスク：odpt/fetch.py の実装

以下の仕様で実装すること。

```python
# odpt/fetch.py
"""
ODPT API v4 からバス関連マスタデータを取得し、
data/raw/ にJSONファイルとして保存するスクリプト。
一度実行したら、ダイヤ改正まで再実行不要。
"""

import os, json, requests
from datetime import datetime
from pathlib import Path

ODPT_BASE = "https://api.odpt.org/api/4/"
ENDPOINTS = {
    "BusTimetable":     "odpt:BusTimetable",
    "BusroutePattern":  "odpt:BusroutePattern",
    "BusstopPole":      "odpt:BusstopPole",
    "Bus":              "odpt:Bus",
}
OPERATOR = "odpt.Operator:Toei"  # 対象事業者（東急は odpt.Operator:Tokyu 等に変更）

def fetch_all(api_key: str, output_dir: str = "data/raw"):
    Path(output_dir).mkdir(parents=True, exist_ok=True)
    date_str = datetime.now().strftime("%Y%m%d")
    
    for name, odpt_type in ENDPOINTS.items():
        params = {
            "acl:consumerKey": api_key,
            "@type": odpt_type,
            "odpt:operator": OPERATOR,
        }
        resp = requests.get(ODPT_BASE + "objects/", params=params)
        resp.raise_for_status()
        data = resp.json()
        
        out_path = Path(output_dir) / f"{name.lower()}_{date_str}.json"
        with open(out_path, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        print(f"[fetch] {name}: {len(data)}件 → {out_path}")
```

**注意点**:
- APIキーは環境変数 `ODPT_API_KEY` から読み込むこと（ハードコード禁止）
- `dct:valid` フィールドを確認し、キャッシュが有効な場合は再取得しない（`odpt/cache.py` 参照）
- 目黒営業所のルート（黒01, 黒02, 東98, 渋71, 渋72, さんまバス）のみに絞り込むフィルタを追加すること

### 3.2 タスク：odpt/normalize.py の実装

JSON-LDをSQLiteとParquetに変換する。

```python
# odpt/normalize.py
"""
data/raw/ の生JSONをフラット化し、
SQLite（data/processed/trips.db）と
Parquet（data/processed/trips.parquet）に変換する。
"""

import json
import sqlite3
import pandas as pd
from pathlib import Path

def flatten_timetable_item(timetable_obj: dict, timetable_id: str) -> list[dict]:
    """
    odpt:BusTimetableのネスト配列 odpt:busTimetableObject を展開して
    フラットな辞書のリストに変換する。
    """
    items = timetable_obj.get("odpt:busTimetableObject", [])
    result = []
    for idx, item in enumerate(items):
        result.append({
            "timetable_id": timetable_id,
            "seq": idx,
            "busstop_pole": item.get("odpt:busstopPole"),
            "departure_time": item.get("odpt:departureTime"),
            "arrival_time": item.get("odpt:arrivalTime"),
            "is_non_step": int(item.get("odpt:isNonStep", False)),
            "note": item.get("odpt:note"),
        })
    return result

def build_sqlite(raw_dir: str = "data/raw", db_path: str = "data/processed/trips.db"):
    """全JSONをSQLiteに正規化してロード"""
    Path(db_path).parent.mkdir(parents=True, exist_ok=True)
    con = sqlite3.connect(db_path)
    con.execute("PRAGMA journal_mode=WAL")  # 並行アクセス改善
    
    # --- BusTimetable テーブル ---
    con.execute("""
        CREATE TABLE IF NOT EXISTS bus_timetable (
            id TEXT PRIMARY KEY,
            operator TEXT,
            busroute TEXT,
            pattern TEXT,
            calendar TEXT,
            dc_date TEXT
        )
    """)
    con.execute("CREATE INDEX IF NOT EXISTS idx_timetable_route ON bus_timetable(busroute)")
    
    # --- BusTimetableItem テーブル（展開済みネスト） ---
    con.execute("""
        CREATE TABLE IF NOT EXISTS bus_timetable_item (
            timetable_id TEXT,
            seq INTEGER,
            busstop_pole TEXT,
            departure_time TEXT,
            arrival_time TEXT,
            is_non_step INTEGER,
            note TEXT
        )
    """)
    con.execute("CREATE INDEX IF NOT EXISTS idx_item_tid ON bus_timetable_item(timetable_id)")
    
    # --- BusstopPole テーブル ---
    con.execute("""
        CREATE TABLE IF NOT EXISTS busstop_pole (
            id TEXT PRIMARY KEY,
            title TEXT,
            operator TEXT,
            lat REAL,
            lon REAL
        )
    """)
    
    # データロード
    for json_file in Path(raw_dir).glob("bustimetable_*.json"):
        with open(json_file) as f:
            data = json.load(f)
        rows = [{
            "id": obj.get("owl:sameAs"),
            "operator": obj.get("odpt:operator"),
            "busroute": obj.get("odpt:busroute"),
            "pattern": obj.get("odpt:busroutePattern"),
            "calendar": obj.get("odpt:calendar"),
            "dc_date": obj.get("dc:date"),
        } for obj in data]
        pd.DataFrame(rows).to_sql("bus_timetable", con, if_exists="append", index=False)
        
        items = []
        for obj in data:
            items.extend(flatten_timetable_item(obj, obj.get("owl:sameAs")))
        pd.DataFrame(items).to_sql("bus_timetable_item", con, if_exists="append", index=False)
    
    con.commit()
    con.close()
    print(f"[normalize] SQLite構築完了: {db_path}")

def export_parquet(db_path: str = "data/processed/trips.db",
                   out_path: str = "data/processed/trips.parquet"):
    """SQLite → Parquet変換（高速読み込み用）"""
    con = sqlite3.connect(db_path)
    df = pd.read_sql_query("""
        SELECT t.busroute, t.pattern, t.calendar,
               i.seq, i.busstop_pole, i.departure_time, i.is_non_step
        FROM bus_timetable t
        JOIN bus_timetable_item i ON t.id = i.timetable_id
        ORDER BY t.busroute, t.pattern, t.calendar, i.seq
    """, con)
    con.close()
    df.to_parquet(out_path, compression="snappy")
    print(f"[normalize] Parquetエクスポート完了: {out_path} ({len(df)}行)")
```

**ALNSの使い方：最小構成**:

```python
from alns import ALNS
from alns.accept import HillClimbing, SimulatedAnnealing, LateAcceptanceHillClimbing
from alns.select import RouletteWheel, RandomSelect
from alns.stop import MaxIterations, MaxRuntime
import numpy.random as rnd

# 1. SolutionStateクラスを定義する（必須）
class BusScheduleState:
    """
    バス車両スケジューリングの解状態。
    研究の21制約MILPから生成した初期解を受け取り、
    Destroy/Repairで近傍を探索する。
    """
    def __init__(self, assignment: dict):
        # assignment: {trip_id: vehicle_id} の辞書
        self.assignment = assignment
    
    def objective(self) -> float:
        """目的関数値（小さいほど良い）を返す"""
        # 例：総コスト（電力費 + デマンド料金 + ICE燃料費）
        return compute_total_cost(self.assignment)
    
    def copy(self):
        return BusScheduleState(self.assignment.copy())


# 2. Destroyオペレーター（解の一部を破壊）
def random_destroy(state: BusScheduleState, rnd_state: rnd.RandomState):
    """ランダムにq本のトリップ割り当てを削除"""
    destroyed = state.copy()
    q = max(1, len(state.assignment) // 10)  # 10%を削除
    trips_to_remove = rnd_state.choice(
        list(state.assignment.keys()), size=q, replace=False
    )
    for trip in trips_to_remove:
        destroyed.assignment.pop(trip)
    return destroyed

def charge_window_destroy(state: BusScheduleState, rnd_state: rnd.RandomState):
    """SOC制約を違反しているトリップ周辺を削除（問題特化Destroyオペレーター）"""
    destroyed = state.copy()
    # SOC制約違反のトリップを特定し、その前後の割り当てを削除
    violations = find_soc_violations(destroyed.assignment)
    for trip in violations[:5]:
        destroyed.assignment.pop(trip, None)
    return destroyed


# 3. Repairオペレーター（削除された部分を修復）
def greedy_repair(state: BusScheduleState, rnd_state: rnd.RandomState):
    """削除されたトリップをSOCと充電ウィンドウを考慮してグリーディに再割り当て"""
    repaired = state.copy()
    unassigned = [t for t in ALL_TRIPS if t not in repaired.assignment]
    for trip in sorted(unassigned, key=lambda t: t.departure_time):
        best_vehicle = find_feasible_vehicle(trip, repaired.assignment)
        if best_vehicle:
            repaired.assignment[trip] = best_vehicle
    return repaired


# 4. ALNSを実行する
alns = ALNS(rnd.RandomState(42))

# Destroyオペレーターを登録
alns.add_destroy_operator(random_destroy)
alns.add_destroy_operator(charge_window_destroy)

# Repairオペレーターを登録
alns.add_repair_operator(greedy_repair)

# 初期解（MILPから生成）
initial_solution = BusScheduleState(milp_initial_assignment)

# 受理基準（Simulated Annealing）
accept = SimulatedAnnealing(
    start_temperature=1.0,
    end_temperature=0.01,
    step=0.9999,
)

# オペレーター選択（ルーレットホイール）
select = RouletteWheel(
    scores=[5, 2, 1, 0.5],  # best/better/accepted/rejected の報酬
    decay=0.8,
    num_destroy=len(alns.destroy_operators),
    num_repair=len(alns.repair_operators),
)

# 停止基準（最大イテレーション）
stop = MaxIterations(max_iterations=10000)

# 実行
result = alns.solve(initial_solution, select, accept, stop)

print(f"初期解コスト: {initial_solution.objective()}")
print(f"最終解コスト: {result.best_state.objective()}")
result.plot_objectives()  # 収束グラフを表示
```

**本研究でのDestroyオペレーター設計指針**:
- `RandomTripDestroy`: ランダムにq本のトリップ割り当てを削除
- `SOCViolationDestroy`: SOC制約（State of Charge）違反箇所周辺を削除
- `PeakDemandDestroy`: デマンド料金が高い時間帯の充電割り当てを削除
- `ICEClusterDestroy`: ICE車両の連続稼働を集中削除（電化促進）

**本研究でのRepairオペレーター設計指針**:
- `GreedyChargeRepair`: SOCフィジビリティを確認しながらグリーディに充電スロットを埋める
- `PVAlignedRepair`: PV発電量プロファイルに合わせて充電時間帯を選択
- `RegretRepair`: regret-based insertion（最良・次善の差が大きいトリップを優先）

**PuLPを使ったMILP定式化の基本**:

```python
import pulp

prob = pulp.LpProblem("BusChargeSchedule", pulp.LpMinimize)

# 変数定義例
x = {(v, t): pulp.LpVariable(f"x_{v}_{t}", cat="Binary")
     for v in vehicles for t in trips}

# 目的関数（電力費 + デマンド料金 + ICE燃料費）
prob += pulp.lpSum([
    electricity_cost[s] * charge_power[v][s] * x[v, t]
    for v in bev_vehicles for t in trips for s in slots
]) + demand_charge * peak_demand

# 制約：各トリップに車両を1台割り当て
for t in trips:
    prob += pulp.lpSum([x[v, t] for v in vehicles]) == 1

# SOC制約
for v in bev_vehicles:
    for t in trips:
        prob += soc[v][t] >= soc_min[v]
        prob += soc[v][t] <= soc_max[v]

# 求解
prob.solve(pulp.PULP_CBC_CMD(msg=1, timeLimit=300))
print(pulp.LpStatus[prob.status])
```

**本研究への活用ポイント**:

oemof-solphはMILP本体の求解には使わないが、以下の目的で部分的に参照すること。

1. **PVプロファイル生成パターン**: oemof-solphのソース時系列定義（`solph.components.Source`のnominal_capacity設定）を参考に、PV発電量の時系列データ（kW/時間）を入力データとして整形する

import pandas as pd

# 時間インデックス（1時間刻み、24時間）
time_index = pd.date_range(start="2025-04-01", periods=25, freq="1h")
energy_system = solph.EnergySystem(timeindex=time_index, infer_last_interval=False)

# バスが在駐している時間帯のマスク（0=走行中、1=デポ在駐）
at_depot = pd.Series(0.0, index=time_index[:-1])

energy_system.add(bus_electricity)

# PV発電源

    label="PV",
    outputs={bus_electricity: solph.Flow(
        nominal_capacity=200,  # kW
        fix=pv_profile,        # PV発電量プロファイル（正規化済み0-1）
    )}
)
# EVバッテリー（バスのSOCモデル）
ev_battery = solph.components.GenericStorage(
    label="BEV_battery",
    nominal_storage_capacity=314,   # kWh (BYD K8 2.0)
    inputs={bus_electricity: solph.Flow(
        nominal_capacity=60,         # 充電器出力 kW
        max=at_depot,                # デポ在駐時のみ充電可
    )},
    initial_storage_level=0.8,
    min_storage_level=0.2,
    max_storage_level=1.0,
)

### 5.2 models/pipeline.py の実装指示

```python
# models/pipeline.py
"""
MILP初期解 → ALNS改善のメインパイプライン
"""
from pathlib import Path
import pandas as pd

from models.milp.formulation import build_milp
from models.milp.solve import solve_milp
from models.alns.state import BusScheduleState
from models.alns.operators import (
    random_destroy, charge_window_destroy, peak_demand_destroy,
    greedy_repair, pv_aligned_repair,
)
from alns import ALNS
from alns.accept import SimulatedAnnealing
from alns.select import RouletteWheel
from alns.stop import MaxIterations
import numpy.random as rnd


def run_pipeline(
    trips_csv: str = "data/input/trips_meguro.csv",
    pv_csv: str = "data/input/pv_profile.csv",
    milp_time_limit: int = 300,    # 秒
    alns_iterations: int = 10000,
    case_name: str = "CaseA",
    seed: int = 42,
):
    # Step 1: 入力データ読み込み
    trips_df = pd.read_csv(trips_csv)
    pv_df = pd.read_csv(pv_csv)
    
    # Step 2: MILP初期解（制限時間内での実行可能解）
    milp_model = build_milp(trips_df, pv_df)
    initial_assignment = solve_milp(milp_model, time_limit=milp_time_limit)
    print(f"[MILP] 初期解コスト: {initial_assignment.cost:.2f}円")
    
    # Step 3: ALNS改善
    alns_solver = ALNS(rnd.RandomState(seed))
    alns_solver.add_destroy_operator(random_destroy)
    alns_solver.add_destroy_operator(charge_window_destroy)
    alns_solver.add_destroy_operator(peak_demand_destroy)
    alns_solver.add_repair_operator(greedy_repair)
    alns_solver.add_repair_operator(pv_aligned_repair)
    
    init_state = BusScheduleState(initial_assignment)
    
    result = alns_solver.solve(
        init_state,
        select=RouletteWheel([5, 2, 1, 0.5], 0.8,
                             len(alns_solver.destroy_operators),
                             len(alns_solver.repair_operators)),
        accept=SimulatedAnnealing(1.0, 0.01, 0.9999),
        stop=MaxIterations(alns_iterations),
    )
    
    print(f"[ALNS] 最終解コスト: {result.best_state.objective():.2f}円")
    print(f"[ALNS] 改善率: {(init_state.objective() - result.best_state.objective()) / init_state.objective() * 100:.1f}%")
    
    # Step 4: 結果保存
    out_path = Path("analysis") / f"result_{case_name}.csv"
    result.best_state.to_csv(out_path)
    result.plot_objectives(fname=f"analysis/convergence_{case_name}.png")
    
    return result
```
